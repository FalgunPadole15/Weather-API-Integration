/* ── WMO CODE → EMOJI + DESCRIPTION ── */
function wmoInfo(code) {
  const map = {
    0:  ['☀️','Clear Sky'],
    1:  ['🌤','Mainly Clear'], 2: ['⛅','Partly Cloudy'], 3: ['☁️','Overcast'],
    45: ['🌫','Foggy'],        48: ['🌫','Icy Fog'],
    51: ['🌦','Light Drizzle'],53: ['🌦','Moderate Drizzle'],55: ['🌧','Heavy Drizzle'],
    61: ['🌧','Slight Rain'],  63: ['🌧','Moderate Rain'],   65: ['🌧','Heavy Rain'],
    71: ['🌨','Slight Snow'],  73: ['❄️','Moderate Snow'],   75: ['❄️','Heavy Snow'],
    80: ['🌦','Light Showers'],81: ['🌧','Showers'],         82: ['⛈','Heavy Showers'],
    95: ['⛈','Thunderstorm'], 96: ['⛈','Thunderstorm'],99: ['⛈','Heavy Thunderstorm'],
  };
  return map[code] || ['🌡','Unknown'];
}
 
/* ── DAY LABELS ── */
function dayLabel(dateStr, i) {
  if (i === 0) return 'Today';
  if (i === 1) return 'Tomorrow';
  return new Date(dateStr).toLocaleDateString('en-US', { weekday: 'short' });
}
 
/* ── FETCH PIPELINE ── */
async function fetchWeather(city) {
  showLoader(true); showError(''); showGrid(false);
 
  // 1. Geocode
  const geoRes = await fetch(
    `https://geocoding-api.open-meteo.com/v1/search?name=${encodeURIComponent(city)}&count=1&language=en&format=json`
  );
  const geoData = await geoRes.json();
  if (!geoData.results || geoData.results.length === 0) throw new Error(`City "${city}" not found. Try a different spelling.`);
 
  const { name, country, latitude, longitude, admin1 } = geoData.results[0];
 
  // 2. Weather
  const wRes = await fetch(
    `https://api.open-meteo.com/v1/forecast?latitude=${latitude}&longitude=${longitude}` +
    `&current=temperature_2m,apparent_temperature,relative_humidity_2m,wind_speed_10m,` +
    `cloud_cover,precipitation,weather_code` +
    `&hourly=temperature_2m,weather_code&forecast_hours=24` +
    `&daily=weather_code,temperature_2m_max,temperature_2m_min` +
    `&timezone=auto&forecast_days=7`
  );
  const w = await wRes.json();
  if (!w.current) throw new Error('Weather data unavailable for this location.');
 
  // Render
  const [icon, desc] = wmoInfo(w.current.weather_code);
  document.getElementById('r-city').textContent    = name;
  document.getElementById('r-country').textContent = [admin1, country].filter(Boolean).join(', ');
  document.getElementById('r-temp').innerHTML      = `${Math.round(w.current.temperature_2m)}<sup>°C</sup>`;
  document.getElementById('r-desc').textContent    = desc;
  document.getElementById('r-icon').textContent    = icon;
 
  document.getElementById('r-humid').textContent  = `${w.current.relative_humidity_2m}%`;
  document.getElementById('r-wind').textContent   = `${w.current.wind_speed_10m} km/h`;
  document.getElementById('r-feels').textContent  = `${Math.round(w.current.apparent_temperature)}°C`;
  document.getElementById('r-cloud').textContent  = `${w.current.cloud_cover}%`;
  document.getElementById('r-precip').textContent = `${w.current.precipitation} mm`;
  document.getElementById('r-coords').textContent = `${latitude.toFixed(2)}°, ${longitude.toFixed(2)}°`;
 
  // Hourly
  const hourlyEl = document.getElementById('r-hourly');
  hourlyEl.innerHTML = '';
  for (let i = 0; i < 24; i++) {
    const time = w.hourly.time[i].split('T')[1];
    const temp = Math.round(w.hourly.temperature_2m[i]);
    const [hIcon] = wmoInfo(w.hourly.weather_code[i]);
    hourlyEl.innerHTML += `
      <div class="hourly-item">
        <div class="h-time">${time}</div>
        <div class="h-icon">${hIcon}</div>
        <div class="h-temp">${temp}°</div>
      </div>`;
  }
 
  // Forecast
  const fcEl = document.getElementById('r-forecast');
  fcEl.innerHTML = '';
  w.daily.time.forEach((date, i) => {
    const [fIcon] = wmoInfo(w.daily.weather_code[i]);
    fcEl.innerHTML += `
      <div class="forecast-day">
        <div class="fc-day">${dayLabel(date, i)}</div>
        <div class="fc-icon">${fIcon}</div>
        <div class="fc-hi">${Math.round(w.daily.temperature_2m_max[i])}°</div>
        <div class="fc-lo">${Math.round(w.daily.temperature_2m_min[i])}°</div>
      </div>`;
  });
 
  showLoader(false); showGrid(true);
}
 
/* ── UI HELPERS ── */
function showLoader(v) { document.getElementById('loader').style.display = v ? 'flex' : 'none'; }
function showGrid(v)   { document.getElementById('grid').style.display   = v ? 'flex' : 'none'; }
function showError(msg){ const el = document.getElementById('error-msg'); el.textContent = msg; el.style.display = msg ? 'block' : 'none'; }
 
/* ── EVENTS ── */
document.getElementById('search-btn').addEventListener('click', run);
document.getElementById('city-input').addEventListener('keydown', e => { if (e.key === 'Enter') run(); });
 
function run() {
  const city = document.getElementById('city-input').value.trim();
  if (!city) { showError('Please enter a city name.'); return; }
  fetchWeather(city).catch(err => { showLoader(false); showError(err.message); });
}
